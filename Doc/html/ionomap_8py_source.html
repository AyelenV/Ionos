225  ## Genera plot pcolormesh de z-data interpolada con scipy.interpolate.SmoothBivariateSpline.
226  # @param ax : Instancia tipo Axes (matplotlib.axes.Axes)
227  # @param smooth : float : coeficiente de suavizado.
228  # @param finegrid_size : [float, float] : dx,dy de grillado fino de interpolacion
229  # @return pc : instancia de clase 'matplotlib.collections.QuadMesh'
230  # @note Preferentemente 0 < smooth < 0.5
231  def SBS(self, ax, smooth, finegrid_size):
232 
233  # Elimino data con valores 'nan'
234  fillbin = np.where(~np.isnan(self.param))
235  # Convierto arreglos 2D a 1D
236  xf = self.x[fillbin].ravel()
237  yf = self.y[fillbin].ravel()
238  zf = self.param[fillbin].ravel()
239  # print zf.shape, xf.shape, yf.shape
240 
241  # Grilla fina regularmente espaciada
242  dx, dy = finegrid_size[0], finegrid_size[1]
243  xi = np.arange(xf.min(), xf.max(), dx)
244  yi = np.arange(yf.min(), yf.max(), dy)
245 
246  # Use SmoothBivariateSpline interpolation
247  z2sum = np.sum( zf**2 )
248  kx, ky = 3, 3
249  # Spline s = smooth * z2sum, see note
250  # s is a target for sum (Z() - spline())**2 ~ Ndata and Z**2;
251  # smooth is relative, s absolute
252  '''
253  NOTE: BivariateSpline y and x to be passed in this order (opposite to that of interp2d.)
254  This issue is related to the way that meshgrid is indexed which is based on the conventions of MATLAB.
255  '''
256 
257  fit = SmoothBivariateSpline(yf, xf, zf, kx=kx, ky=ky, s=smooth*z2sum)
258  Zspline = fit(yi,xi)
259  # Plot
260  pc = ax.pcolormesh(xi, yi, Zspline, cmap=self.cm, vmax=self.vmax, vmin=self.vmin)
261  ax.set_title(self.par_name + ' - ' + self.time + ' SBSpline '+ str(smooth))
262 
263  return pc
