45  Calcula un estadistico de la data original, dividida en bins de tamanno ajustable, binsize = [dx,dy].

48  '''
49

52  nby = int(abs(self.y.max()- self.y.min())/binsize[1])
53  print 'bins number: ', nbx, nby

55  '''
56  ATENCION: argumentos de scipy.stats.binned_statistic_2d: y,y,z en ese orden!

58  '''

179 
180  '''
181  Genera objeto plot tipo pcolormesh de z-data interpolada con scipy.interpolate.griddata.
182  El tipo de interpolacion y la correspondiente grilla de interpolacion, son ajustadas por usuario.
183  '''
184 
185  # Tipos de interpolacion disponibles para scipy.griddata
186  interpol_list = ['nearest', 'linear', 'cubic']
187 
188  # Elimino data con valores 'nan'
189  fillbin = np.where(~np.isnan(self.param))
190  # Convierto arreglos 2D a 1D
191  xf = self.x[fillbin].ravel()
192  yf = self.y[fillbin].ravel()
193  zf = self.param[fillbin].ravel()
194 
195  # Grilla fina regularmente espaciada
196  dx, dy = finegrid_size[0], finegrid_size[1]
197  xi = np.arange(xf.min(), xf.max(), dx)
198  yi = np.arange(yf.min(), yf.max(), dy)
199  Xig, Yig = np.meshgrid(xi, yi)
200 
201  if interpol in interpol_list:
202  # Interpolo z-data con scipy.interpolate.griddata
203  Zi = griddata((xf, yf), zf, (Xig, Yig), method=interpol)
204  # print Zi.shape, xi.shape, yi.shape

207  ax.set_title(self.par_name + ' - ' + self.time + ' griddata: '+ interpol)

214  def SBS(self, ax, smooth, finegrid_size):
215 
216  '''
217  Genera objeto plot tipo pcolormesh de z-data interpolada con scipy.interpolate.SmoothBivariateSpline.
218  La grilla fina de interpolacion y el coeficiente de suavizado 'smooth', son ajustadas por usuario.
219  '''
220 
221  # Elimino data con valores 'nan'
222  fillbin = np.where(~np.isnan(self.param))
223  # Convierto arreglos 2D a 1D
224  xf = self.x[fillbin].ravel()
225  yf = self.y[fillbin].ravel()
226  zf = self.param[fillbin].ravel()
227  # print zf.shape, xf.shape, yf.shape
228 
229  # Grilla fina regularmente espaciada
230  dx, dy = finegrid_size[0], finegrid_size[1]
231  xi = np.arange(xf.min(), xf.max(), dx)
232  yi = np.arange(yf.min(), yf.max(), dy)
233 
234  # Use SmoothBivariateSpline interpolation
235  z2sum = np.sum( zf**2 )
236  kx, ky = 3, 3
237  # Spline s = smooth * z2sum, see note
238  # s is a target for sum (Z() - spline())**2 ~ Ndata and Z**2;
239  # smooth is relative, s absolute
240  '''
241  NOTE: BivariateSpline y and x to be passed in this order (opposite to that of interp2d.)
242  This issue is related to the way that meshgrid is indexed which is based on the conventions of MATLAB.
243  '''
244 
245  fit = SmoothBivariateSpline(yf, xf, zf, kx=kx, ky=ky, s=smooth*z2sum)
246  Zspline = fit(yi,xi)
247  # Plot
248  pc = ax.pcolormesh(xi, yi, Zspline, cmap=self.cm, vmax=self.vmax, vmin=self.vmin)
249  ax.set_title(self.par_name + ' - ' + self.time + ' SBSpline '+ str(smooth))
250 
251  return pc
